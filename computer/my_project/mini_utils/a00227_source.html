21  * @brief [0, last-1)范围中的向量元素是一个堆.
22  * 将v[last-1]元素插入到堆中, 使[0, last)是一个堆.
23  * 使用函数对象comp进行比较.
24  *
25  * @tparam T 堆中元素类型
26  * @tparam Compare 堆中元素比较的函数对象的类型
27  * @param v 存放堆元素的数组
28  * @param last 插入堆中元素的下标的下一个位置
29  * @param comp 元素比较的函数对象
30  *
31  * @note 如果Compare为less<T>, 那么得到的是一个最大堆.
32  * 如果Compare为greater<T>, 那么得到的是一个最小堆.
33  */
34 template <typename T, typename Compare=std::less<T>>
35 void push_heap(std::vector<T> &v, int last, Compare comp=Compare())
36 {
37  // 假设新项的位置为v[last-1], 而且, v[0]到v[last2]的元素以
38  // 堆的顺序排列
39  int currentPos, parentPos;
40  T target;
41 
42  // currentPos是用于遍历父节点路径的索引.
43  // target是v[last-1]值, 需要在路径中重新定位
44  currentPos = last-1;
45  parentPos = (currentPos-1)/2;
46  target = v[last-1];
47 
48  // 遍历父结点的路径, 直到根
49  while (currentPos != 0)
50  {
51  // 比较target和父结点值
52  if (comp(v[parentPos], target))
53  {
54  // 将数据从父结点位置移动到当前位置.
55  // 将表示当前位置的索引更新为父结点位置,
56  // 计算下一个父结点位置
57  v[currentPos] = v[parentPos];
58  currentPos = parentPos;
59  parentPos = (currentPos-1)/2;
60  }
61  else
62  // 如果!comp(parentvalue, target), 则满足堆条件
63  break;
64  }
65  // 正确位置已被发现, 将其赋值为target
66  v[currentPos] = target;
67 }

69 /**
70  * @brief 在索引范围[first, last)中, 下移向量元素v[first]
71  *
72  * @tparam T 堆中元素类型
73  * @tparam Compare 堆中元素比较的函数对象的类型
74  * @param v 存放堆元素的数组
75  * @param first 需要下移向量元素的索引
76  * @param last 索引范围最后一个元素的下一个位置
77  * @param comp 元素比较的函数对象
78  *
79  * @note 如果Compare为less<T>, 那么得到的是一个最大堆.
80  * 如果Compare为greater<T>, 那么得到的是一个最小堆.
81  */
82 template <typename T, typename Compare=std::less<T>>
83 void adjust_heap(std::vector<T> &v, int first, int last, Compare comp=Compare())
84 {
85  int currentPos, childPos;
86  T target;
87 
88  // target是v[first]值, 需要在路径中重新定位
89  currentPos = first;
90  target = v[first];
91 
92  // 计算左子结点的索引, 开始向下扫描子结点的路径, 在表尾(last)终止,
93  // 或者, 在找到target的位置时终止
94  childPos = 2 * currentPos + 1;
95  while (childPos <= last-1)
96  {
97  // 右子结点的索引为childPos+1. 比较两个子结点,
98  // 如果comp(v[childPos], v[childPos+1])为真, 则改变childPos
99  if ((childPos+1 <= last-1) &&
100  comp(v[childPos], v[childPos+1]))
101  childPos = childPos + 1;
102 
103  // 将所选子结点与target进行比较
104  if (comp(target, v[childPos]))
105  {
106  // comp(target, selected child)为真, 将所选子结点移到父结点;
107  // 现在, 所选子结点的位置空出
108  v[currentPos] = v[childPos];
109 
110  // 更新索引, 以便继续扫描
111  currentPos = childPos;
112  childPos = 2 * currentPos + 1;
113  }
114  else
115  // target属于currentPos指定的位置.
116  break;
117  }
118  v[currentPos] = target;
119 }

123  * 使索引范围[0, last-1)中的元素堆化. 使用函数对象comp进行比较.
124  *

126  * @tparam Compare 堆中元素比较的函数对象的类型
127  * @param v 存放堆元素的数组
128  * @param last 堆中最后一个元素的下一个位置
129  * @param comp 元素比较的函数对象
130  */
131 template <typename T, typename Compare=std::less<T>>
132 void pop_heap(std::vector<T> &v, int last, Compare comp=Compare())
133 {
134  T temp;
135 
136  // 交换堆中的第1个元素和最后一个元素
137  temp = v[0];
138  v[0] = v[last-1];
139  v[last-1] = temp;
140 
141  // 在[0, last-1)范围中下移根结点
142  adjust_heap(v, 0, last-1, comp);
143 }

146  * @brief 将向量元素安排到堆中. 使用函数对象comp进行比较
147  *
148  * @tparam T 堆中元素类型
149  * @tparam Compare 堆中元素比较的函数对象的类型
150  * @param v 存放堆元素的数组
151  * @param comp 元素比较的函数对象
152  */
153 template <typename T, typename Compare=std::less<T>>
154 void make_heap(std::vector<T> &v, Compare comp=Compare())
155 {
156  int heapPos, lastPos;
157 
158  // 从0开始, 并恢复堆的顺序
159  lastPos = v.size();
160  heapPos = (lastPos - 2)/2;
161 
162  // 从最下层的非叶子结点开始堆化, 一直处理到根结点
163  while (heapPos >= 0)
164  {
165  adjust_heap(v, heapPos, lastPos, comp);
166  heapPos--;
167  }
168 }

178 template <typename T, typename Compare=std::less<T>>
179 void sort_heap(std::vector<T> &v, Compare comp=Compare())
180 {
181  // 确定元素v[n-1] ... v[1]的迭代
182  for (int lastPos = v.size(); lastPos > 1; lastPos--) {
183  // 调用pop_heap()将下一个最大结点移到v[n-1]
184  pop_heap(v, lastPos, comp);
185  }
