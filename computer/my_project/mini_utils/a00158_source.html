23 /**
24  * @brief 维护一个包含数据类型T元素的优先级队列,
25  * 优先级队列的keyList由外部维护,
26  * 优先级队列只维护keyList的索引的堆,
27  * 堆化过程只交换索引, 不交换外部实际元素,
28  * 使用类型Compare的比较函数对象
29  *
30  * @tparam T 元素类型
31  * @tparam Compare 比较函数对象的类型
32  *
33  * @note 当Compare为std::greater(大于)的时候, 数值大的元素优先级高;
34  * 当Compare为std::less(小于)的时候, 数值小的元素优先级高;
35  * 这点与std::priority_queue正好相反.
36  */
37 template <class T, typename Compare = std::greater<T>>
38 class PriorityQueueRef {
39 private:
40  std::vector<int> pqList_; // 优先级队列, 存放的元素是keyList_中元素的索引
41  Compare comp_; // 比较使用的函数对象
42  std::unordered_map<int, int> key2pqMap_; // 从keyList_中元素索引, 转换成pqList_中元素索引
43  const std::vector<T> &keyList_; // 存放用于实际计算优先级的key列表(这里是引用, 实际队列有构造函数传入)
44 
45  void exch(int i, int j)
46  {
47  // 交换pqList_[i]和pqList_[j]
48  auto tmp = pqList_[i];
49  pqList_[i] = pqList_[j];
50  pqList_[j] = tmp;
51 
52  // 更新keyList索引到pqList索引的映射
53  key2pqMap_[pqList_[i]] = i;
54  key2pqMap_[pqList_[j]] = j;
55  }
56 
57  // 向根结点方向调整堆
58  void bubbleUp(int currentPos)
59  {
60  int parentPos = (currentPos-1)/2;
61  while (currentPos > 0) {
62  if (comp_(keyList_[pqList_[currentPos]], keyList_[pqList_[parentPos]])) {
63  exch(currentPos, parentPos);
64  currentPos = parentPos;
65  parentPos = (currentPos-1)/2;
66  } else {
67  break;
68  }
69  }
70  }
71 
72  // 向叶子结点方向调整堆
73  void siftDown(int currentPos, int lastPos)
74  {
75  int childPos = 2*currentPos+1;
76  while (childPos < lastPos) {
77  if ((childPos+1 < lastPos) && comp_(keyList_[pqList_[childPos+1]], keyList_[pqList_[childPos]]))
78  childPos = childPos + 1;
79 
80  if (comp_(keyList_[pqList_[childPos]], keyList_[pqList_[currentPos]])) {
81  exch(currentPos, childPos);
82  currentPos = childPos;
83  childPos = 2*currentPos+1;
84  } else {
85  break;
86  }
87  }
88  }

112  /**
113  * @brief 将keyList[idxOfKeyList]插入到优先级队列中
114  *
115  * @param idxOfKeyList 被插入元素在keyList中的下标
116  */
117  void push(int idxOfKeyList)
118  {
119  assert(idxOfKeyList >= 0 && idxOfKeyList < keyList_.size());
120  pqList_.push_back(idxOfKeyList);
121  int i = pqList_.size()-1;
122  key2pqMap_[idxOfKeyList] = i;
123  bubbleUp(i);
124  }

126  /**
127  * @brief 删除具有最高优先级的元素
128  *
129  * @return 返回优先级最高元素在keyList中的下标
130  */
131  int pop()
132  {
133  assert(!pqList_.empty());
134 
135  // 交换pqList的0和size-1下标元素,
136  // 从pqList的0下标开始heapify pqList
137  int idxOfKeyList = pqList_[0];
138  int lastPos = pqList_.size()-1;
139  exch(0, lastPos);
140  siftDown(0, lastPos);
141 
142  // pop last element of pqList
143  pqList_.pop_back();
144  key2pqMap_.erase(idxOfKeyList);
145  return idxOfKeyList;
146  }

149  * @brief 返回具有最高优先级的元素
150  *
151  * @return 返回具有最高优先级的元素在keyList中的下标
152  */
153  int top()
154  {
155  assert(!pqList_.empty());
156  return pqList_[0];
157  }

159  /**

161  *
162  * @param idxOfKeyList 被更新优先级的元素在keyList中的下标
163  */
164  void update(int idxOfKeyList)
165  {
166  assert(key2pqMap_.count(idxOfKeyList) && key2pqMap_[idxOfKeyList] < pqList_.size());
167  bubbleUp(key2pqMap_[idxOfKeyList]);
168  siftDown(key2pqMap_[idxOfKeyList], pqList_.size());
169  }
